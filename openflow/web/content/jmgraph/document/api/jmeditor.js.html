jmEditor.prototype.addCell = function(option) {
	option = option || {};
	option.graph = this.graph;
	option.editor = this;
	if(!this.isEnabled()) {
		option.resizable = false;
	}	
	var cell = new jmCell(option);
	//å¦æå·²è®¾å®æ ¡éªèç¹ï¼åééè¿æ ¡éªæå¯æ·»å 
	if(this.validCell && !this.validCell(cell)) {
		return;
	}
	var self = this;
	//çæ§å¯¹è±¡éæ©äºä»¶
	cell.on('select',function(s) {
		self.emit('select',this,s);
	});
	
	//åç´ ç§»å¨æ¶éæ°å¤çå¶å¤§å°
	cell.on('move',function() {
		resetEditorSize(this);
	});
	cell.id = option.id || this.maxId();
	this.cells.add(cell);
	cell.add();

	/**
	 * æ ¹æ®ææ°åç´ éæ°è®¡ç®ç»å¸å¤§å°
	 *
	 * @method resetEditorSize
	 * @private
	 */
	function resetEditorSize(target) {
		var p = target.position();
		var maxw = p.x + target.width();
		var maxh = p.y + target.height();
		if(maxw >= target.graph.canvas.width || maxh >= target.graph.canvas.height) {
			target.graph.redraw(Math.max(maxw,target.graph.canvas.width) + 10,Math.max(maxh,target.graph.canvas.height) + 10);
		}
	}
	resetEditorSize(cell);
	return cell;
}

jmEditor.prototype.save = function() {
	if(!this.isEnabled()) return;//ä¸å¯ç¼è¾æ¶ï¼ç´æ¥éåº
	if(!this.caches) this.caches = [];
	//å½åç¼å­ç´¢å¼ä¸æ¯æåä¸ä¸ªæ¶ï¼è¡¨ç¤ºæ­£å¨åæ¤æ¶ææ¢å¤æä½

	if(this.cacheIndex < this.caches.length - 1) {

/**
 * è½¬ä¸ºjsonå¯¹è±¡
 * ç»å¸å®æ´jsonæè¿°
 * 
 * @method toJSON
 * @for jmEditor
 * @return {object} å½åæè¿°json
 */
jmEditor.prototype.toJSON = function() {
	var json = {cells:[]};
	this.cells.each(function(i,cell) {		
		var c = {outs:[]};
		c.id = cell.id;
		c.position = cell.position();
		c.width = cell.width();
		c.height = cell.height();
		c.style = cell.styleName || cell.style;
		c.value = cell.value();
		if(cell.connects) {
			cell.connects.each(function(j,cn) {
				if(cn.from == cell) {
					c.outs.push(
						{
							to:cn.to.id,
							value:cn.value(),
							from:cell.id
						});
				}
			});
		}
		json.cells.push(c);
	}) 
	return JSON.stringify(json);
}

/**
 * ä»jsonä¸­æ¢å¤å¾
 * æ ¹æ®jsonæè¿°æ¢å¤åå¾
 *
 * @method fromJSON
 * @for jmEditor
 * @param {string/object} json æè¿°json
 * @param {boolean} [s=false] å½åæ¢å¤æ¯å¦è®°å½ç¶æ trueè¡¨ç¤ºè®°å½ï¼falseä¸è®°å½
 */
jmEditor.prototype.fromJSON = function(json,s) {
	if(!json) return;
	if(typeof json === 'string') {
		json = JSON.parse(json);
	}
	this.clear();
	if(json.cells) {
		//ä»æºä¸­çææµç¨å¾çèç¹
		for(var i in json.cells) {
			var cell = json.cells[i];
			this.addCell(cell);			
		}
		//å¾ªç¯ææèç¹ï¼è¿æ¥å½¼æ­¤çè¿çº¿
		for(var i in json.cells) {
			var cell = json.cells[i];
			if(cell.outs) {
				//è·åè¿æ¥èµ·å§èç¹
				var cur = this.getCell(cell.id);
				if(!cur) continue;
				for(var j in cell.outs) {
					var o = cell.outs[j];
					if(o.to) {
						//è·åç®æ èç¹å¯¹è±¡
						var target = this.getCell(o.to);
						if(target) {
							cur.connect(target,o.value);//è¿æ¥
						}
					}
				}
			}	
		}
		this.graph.refresh();
	}
	if(s !== true) {
		this.save();
	}	
}
